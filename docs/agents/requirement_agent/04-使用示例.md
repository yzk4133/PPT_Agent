## 🔧 3. 高级用法

### 3.1 批量解析

```python
import asyncio
from backend.agents.core.requirements import RequirementParserAgent

async def batch_parse():
    agent = RequirementParserAgent()

    inputs = [
        "生成一份关于AI的PPT，15页",
        "帮我做一个Q3销售报告，商务风格",
        "Create a presentation about climate change, 20 pages",
        "学术研究PPT，关于深度学习，25页"
    ]

    # 并发解析所有输入
    tasks = [agent.parse(inp) for inp in inputs]
    results = await asyncio.gather(*tasks)

    # 打印结果
    for i, result in enumerate(results):
        print(f"\n结果{i+1}:")
        print(f"  主题: {result.ppt_topic}")
        print(f"  页数: {result.page_num}")
        print(f"  语言: {result.language.value}")
        print(f"  模板: {result.template_type.value}")

asyncio.run(batch_parse())
```

### 3.3 错误处理

```python
from backend.agents.core.requirements import (
    RequirementParserAgent,
    RequirementParseError
)

async def error_handling_example():
    agent = RequirementParserAgent()

    # 示例1: 处理空输入
    try:
        result = await agent.parse("")
    except ValueError as e:
        print(f"❌ 输入为空: {e}")

    try:
        result = await agent.parse("test")

        print(f"❌ 解析失败: {e.message}")

            ppt_topic="默认主题",
            page_num=10,
            language=Language.ZH_CN
        )
        print(f"✅ 使用默认需求: {result.ppt_topic}")

asyncio.run(error_handling_example())
```

## 🔌 5. 集成示例

### 5.1 集成到FastAPI

```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from backend.agents.core.requirements import create_requirement_parser

app = FastAPI()

class ParseRequest(BaseModel):
    user_input: str

class ParseResponse(BaseModel):
    ppt_topic: str
    page_num: int
    language: str
    template_type: str
    scene: str

@app.post("/api/parse", response_model=ParseResponse)
async def parse_requirement(request: ParseRequest):
    """解析用户需求"""
    try:
        agent = create_requirement_parser()
        result = await agent.parse(request.user_input)

        return ParseResponse(
            ppt_topic=result.ppt_topic,
            page_num=result.page_num,
            language=result.language.value,
            template_type=result.template_type.value,
            scene=result.scene.value
        )
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

# 运行: uvicorn main:app --reload
```

  -d '{"user_input": "生成一份关于人工智能的PPT，15页"}'

### 5.2 集成到LangGraph工作流

```python
from langgraph.graph import StateGraph, END
from backend.agents.core.requirements import create_requirement_parser
from typing import TypedDict

class PPTGenerationState(TypedDict):
    user_input: str
    structured_requirements: dict

# 创建需求解析节点
async def requirement_node(state: PPTGenerationState) -> PPTGenerationState:
    """需求解析节点"""
    agent = create_requirement_parser()
    requirement = await agent.parse(state["user_input"])

    # 存储到状态中
    state["structured_requirements"] = requirement.model_dump()
    return state

# 构建工作流
workflow = StateGraph(PPTGenerationState)
workflow.add_node("requirement_parser", requirement_node)
workflow.add_edge("requirement_parser", "framework_designer")
workflow.set_entry_point("requirement_parser")

# 编译工作流
app = workflow.compile()
```

import time
import asyncio
from backend.agents.core.requirements import RequirementParserAgent

async def performance_test():

    test_inputs = [
        "生成一份关于人工智能的PPT，15页",
        "帮我做一个学术研究PPT，关于大语言模型，20页",

## 💡 8. 最佳实践

### ✅ DO

```python
# 1. 使用工厂函数创建Agent
agent = create_requirement_parser()

# 2. 批量处理时使用asyncio.gather
tasks = [agent.parse(inp) for inp in inputs]
results = await asyncio.gather(*tasks)

# 3. 处理异常
try:
    result = await agent.parse(user_input)
except ValueError as e:
    print(f"输入错误: {e}")

# 4. 使用枚举而非字符串
if result.language == Language.ZH_CN:
    print("中文")

# 5. 转换结果后再传输
json_data = result.model_dump_json()
```

### ❌ DON'T

```python
# 1. 不要在循环中同步调用
for inp in inputs:
    result = await agent.parse(inp)  # 慢！

# 2. 不要忽略异常
result = await agent.parse(user_input)  # 可能抛出异常

# 3. 不要硬编码枚举值
if result.language == "ZH-CN":  # 应该用 Language.ZH_CN
    print("中文")

# 4. 不要频繁创建Agent
for inp in inputs:
    agent = RequirementParserAgent()  # 每次创建新实例
    result = await agent.parse(inp)
```
