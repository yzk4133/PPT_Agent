## 🎯 1. 核心职责

将用户的自然语言输入转换为结构化的、机器可执行的PPT生成需求。

### 职责定位
- **输入**: 用户的自由文本描述（模板引导式）
- **输出**: PPTRequirement（Pydantic模型）
- **核心价值**: 理解意图 → 提取参数 → 验证补全 → 结构化输出

## 📤 3. 输出字段定义（Pydantic模型）

### 3.1 完整模型定义

```python
from pydantic import BaseModel, Field, field_validator
from typing import List, Optional
from enum import Enum

class Language(str, Enum):
    """语言代码"""
    ZH_CN = "ZH-CN"
    EN_US = "EN-US"

class TemplateType(str, Enum):
    """模板类型"""
    BUSINESS = "business_template"
    ACADEMIC = "academic_template"
    CREATIVE = "creative_template"

class Scene(str, Enum):
    """使用场景"""
    BUSINESS_REPORT = "business_report"
    ACADEMIC_PRESENTATION = "academic_presentation"
    PRODUCT_LAUNCH = "product_launch"
    TRAINING = "training"

class Tone(str, Enum):
    """语调风格"""
    PROFESSIONAL = "professional"
    CASUAL = "casual"
    CREATIVE = "creative"

class PPTRequirement(BaseModel):
    """PPT生成需求模型"""

    # ========== 必须有（解析失败则报错） ==========
    ppt_topic: str = Field(
        ...,
        description="PPT主题",
        min_length=2,
        max_length=100
    )

    page_num: int = Field(
        ...,
        description="PPT页数",
        ge=5,
        le=50
    )

    language: Language = Field(
        ...,
        description="语言：ZH-CN或EN-US"
    )

    # ========== 推荐有（有默认值） ==========
    template_type: TemplateType = Field(
        default=TemplateType.BUSINESS,
        description="模板类型"
    )

    scene: Scene = Field(
        default=Scene.BUSINESS_REPORT,
        description="使用场景"
    )

    tone: Tone = Field(
        default=Tone.PROFESSIONAL,
        description="语调风格"
    )

    # ========== 可选有 ==========
    core_modules: List[str] = Field(
        default_factory=list,
        description="核心模块列表（从主题智能提取）",
        max_length=10
    )

    need_research: bool = Field(
        default=False,
        description="是否需要研究"
    )

    color_scheme: Optional[str] = Field(
        default=None,
        description="色彩方案"
    )

    target_audience: Optional[str] = Field(
        default=None,
        description="目标受众"
    )

    # ========== 验证器 ==========
    @field_validator('ppt_topic')
    @classmethod
    def topic_not_empty(cls, v: str) -> str:
        """主题不能为空"""
        if not v or not v.strip():
            raise ValueError('主题不能为空')
        return v.strip()

    @field_validator('page_num')
    @classmethod
    def page_num_in_range(cls, v: int) -> int:
        """页数必须在5-50之间"""
        if v < 5 or v > 50:
            raise ValueError('页数必须在5-50之间')
        return v

    @field_validator('core_modules')
    @classmethod
    def modules_limit(cls, v: list) -> list:
        """核心模块不超过10个"""
        if len(v) > 10:
            raise ValueError('核心模块不能超过10个')
        return v

    class Config:
        """Pydantic配置"""
        use_enum_values = True  # 返回枚举值而非枚举对象
        extra = "forbid"        # 禁止额外字段（捕获LLM幻觉）
```

## ✅ 4. 解析策略（混合方案）

### 4.1 三层解析架构

```
┌─────────────────────────────────────────────────────┐
│                  用户输入                            │
└──────────────────┬──────────────────────────────────┘
                   │
                   ▼
┌─────────────────────────────────────────────────────┐
│  Layer 1: LLM解析（GPT-4o-mini，10s超时）            │
│  ├─ Few-Shot Prompt                                  │
│  ├─ 输出：JSON字符串                                 │
│  └─ 成功率：~85%                                     │
└──────────────────┬──────────────────────────────────┘
                   │ 成功
                   ▼
┌─────────────────────────────────────────────────────┐
│  Layer 2: JsonOutputParser                           │
│  ├─ 解析JSON字符串 → Dict                            │
│  ├─ 容错：去除markdown包裹                           │
│  └─ 成功率：~95%                                     │
└──────────────────┬──────────────────────────────────┘
                   │ 成功
                   ▼
┌─────────────────────────────────────────────────────┐
│  Layer 3: Pydantic验证                               │
│  ├─ 类型检查、范围检查、枚举检查                     │
│  ├─ 捕获LLM幻觉字段（extra="forbid"）                │
│  ├─ 智能推导：填充默认值                             │
│  └─ 成功率：~99%                                     │
└──────────────────┬──────────────────────────────────┘
                   │ 成功
                   ▼
┌─────────────────────────────────────────────────────┐
│            返回 PPTRequirement 模型                   │
└─────────────────────────────────────────────────────┘
```

### 4.2 错误处理流程

```
任何层级失败
    │
    ├─ LLM超时/失败 → 自动重试1次
    │                 └─ 仍失败 → 规则降级
    │
    ├─ JSON解析失败 → 自动重试1次
    │                 └─ 仍失败 → 规则降级
    │
    ├─ Pydantic验证失败 → 尝试修复
    │                     └─ 修复失败 → 规则降级
    │
    └─ 规则降级 → 返回基本需求
                  └─ 记录失败日志
```

### 4.3 错误处理策略详解

#### 策略1：自动重试机制
```python
MAX_RETRIES = 1  # 最多重试1次

async def parse_with_retry(self, user_input: str) -> dict:
    for attempt in range(MAX_RETRIES + 1):
        try:
            return await self._llm_parse(user_input)
        except Exception as e:
            if attempt == MAX_RETRIES:
                # 最后一次尝试失败，降级
                logger.warning(f"LLM解析失败（尝试{attempt + 1}次）: {e}")
                raise RequirementParseError("LLM解析失败") from e
            logger.info(f"第{attempt + 1}次尝试失败，重试...")
```

#### 策略2：友好错误提示
```python
def _format_validation_error(self, error: ValidationError) -> str:
    """将Pydantic错误转为用户友好的提示"""
    errors = error.errors()
    formatted = []

    for err in errors:
        field = ".".join(str(loc) for loc in err["loc"])
        msg = err["msg"]

        # 翻译常见错误
        if "field required" in msg:
            msg = f"缺少必填字段：{field}"
        elif "ensure this value is greater than" in msg:
            msg = f"{field}的值太小"
        elif "ensure this value is less than" in msg:
            msg = f"{field}的值太大"

        formatted.append(f"• {msg}")

    return "\n".join(formatted)
```

#### 策略3：强制降级
```python
async def parse(self, user_input: str) -> PPTRequirement:
    try:
        # 尝试LLM解析
        return await self._parse_with_llm(user_input)
    except Exception as e:
        logger.error(f"LLM解析完全失败: {e}")
        # 强制降级到规则提取
        fallback_dict = self._fallback_parse(user_input)
        # 确保降级结果也能通过Pydantic验证
        return PPTRequirement(**fallback_dict)
```

#### 策略4：失败日志记录
```python
def _log_parse_failure(
    self,
    user_input: str,
    error: Exception,
    stage: str  # "llm", "json", "pydantic", "fallback"
):
    """记录解析失败案例"""
    logger.error(
        f"[需求解析失败] 阶段: {stage}",
        extra={
            "user_input": user_input[:100],  # 只记录前100字符
            "error_type": type(error).__name__,
            "error_message": str(error),
            "stage": stage
        }
    )
```

## 🔄 6. 降级策略（规则提取）

### 6.1 降级触发条件

| 触发条件 | 处理方式 |
|----------|---------|
| LLM调用超时（>10s） | 立即降级 |
| LLM返回错误 | 重试1次，仍失败则降级 |
| JSON解析失败 | 重试1次，仍失败则降级 |
| Pydantic验证失败 | 尝试修复，仍失败则降级 |

### 6.2 规则提取实现

```python
import re

def _fallback_parse(self, user_input: str) -> dict:
    """
    规则提取（LLM失败时的降级方案）

    保证：永远返回一个可用的需求字典
    """
    result = {
        'ppt_topic': '',
        'page_num': 10,
        'language': 'ZH-CN',
        'template_type': 'business_template',
        'scene': 'business_report',
        'tone': 'professional',
        'core_modules': [],
        'need_research': False
    }

    # 1. 提取主题（前100字符）
    result['ppt_topic'] = user_input[:100].strip()

    # 2. 提取页数（数字 + "页" / "page"）
    page_match = re.search(r'(\d+)\s*[_页页page]', user_input, re.IGNORECASE)
    if page_match:
        page_num = int(page_match.group(1))
        result['page_num'] = max(5, min(50, page_num))

    # 3. 检测语言
    has_chinese = bool(re.search(r'[\u4e00-\u9fff]', user_input))
    result['language'] = 'ZH-CN' if has_chinese else 'EN-US'

    # 4. 识别场景和模板（关键词匹配）
    keywords_map = {
        # 学术
        ('学术', '研究', '论文', 'research'): {
            'template_type': 'academic_template',
            'scene': 'academic_presentation'
        },
        # 创意
        ('创意', '设计', '艺术', 'creative'): {
            'template_type': 'creative_template',
            'tone': 'creative'
        },
        # 产品发布
        ('产品', '发布', '推介', 'launch'): {
            'scene': 'product_launch'
        },
        # 培训
        ('培训', '教程', '教学', 'training'): {
            'scene': 'training'
        }
    }

    for keywords, values in keywords_map.items():
        if any(kw in user_input for kw in keywords):
            result.update(values)
            break

    # 5. 从主题提取核心模块（简单分词）
    if '、' in user_input:
        result['core_modules'] = [m.strip() for m in user_input.split('、') if m.strip()][:5]

    return result
```

### 7.1 继承关系


### 7.2 类定义

```python
from langchain_openai import ChatOpenAI
from langchain_core.output_parsers import JsonOutputParser
from langchain_core.prompts import ChatPromptTemplate

class RequirementParserAgent(BaseAgent):
    """
    需求解析智能体

    职责：
    1. 解析用户自然语言输入
    2. 提取结构化的PPT生成需求
    3. 验证并填充默认值
    4. 提供降级策略

    特性：
    - 继承BaseAgent，获得统一的模型管理
    - 使用混合方案：JsonOutputParser + Pydantic验证
    - Few-Shot Prompt提升理解能力
    - 多层错误处理：自动重试、友好提示、强制降级
    - 超时控制：10秒超时自动降级
    """

    def __init__(
        self,
        model: Optional[ChatOpenAI] = None,
        temperature: float = 0.0,
        agent_name: str = "RequirementParserAgent",
        timeout: int = 10,  # 超时时间（秒）
        max_retries: int = 1  # 最大重试次数
    ):
        """
        初始化

        Args:
            model: LangChain LLM实例（可配置，默认使用配置文件）
            temperature: 温度参数（低温度保证稳定输出）
            agent_name: Agent名称
            timeout: LLM调用超时时间（秒）
            max_retries: 最大重试次数
        """
        super().__init__(
            model=model,
            temperature=temperature,
            agent_name=agent_name
        )

        self.timeout = timeout
        self.max_retries = max_retries

        # 创建解析链
        self.json_parser = JsonOutputParser()
        self.chain = self._setup_chain()

    def _setup_chain(self) -> Runnable:
        """设置LLM解析链"""
        prompt = ChatPromptTemplate.from_template(REQUIREMENT_PARSER_PROMPT)
        return prompt | self.model

    async def parse(self, user_input: str) -> PPTRequirement:
        """
        解析用户输入（主入口）

        Args:
            user_input: 用户自然语言输入

        Returns:
            PPTRequirement: 结构化需求模型

        Raises:
            RequirementParseError: 输入为空或解析完全失败
        """
        # 验证输入
        if not user_input or not user_input.strip():
            raise ValueError("用户输入不能为空")

        logger.info(f"[{self.agent_name}] 开始解析: {user_input[:50]}...")

        try:
            # 尝试LLM解析
            return await self._parse_with_llm(user_input)
        except Exception as e:
            logger.error(f"[{self.agent_name}] LLM解析失败: {e}")
            # 强制降级
            return await self._parse_with_fallback(user_input)

    async def _parse_with_llm(self, user_input: str) -> PPTRequirement:
        """
        使用LLM解析（三层解析架构）

        Layer 1: LLM生成JSON
        Layer 2: JsonOutputParser解析
        Layer 3: Pydantic验证
        """
        for attempt in range(self.max_retries + 1):
            try:
                # Layer 1: LLM生成（带超时控制）
                raw_output = await asyncio.wait_for(
                    self.chain.ainvoke({"user_input": user_input}),
                    timeout=self.timeout
                )

                # Layer 2: JsonOutputParser解析
                parsed_dict = self.json_parser.parse(raw_output)

                # Layer 3: Pydantic验证 + 智能推导
                validated = self._validate_and_infer(parsed_dict)

                logger.info(f"[{self.agent_name}] 解析成功")
                return validated

            except asyncio.TimeoutError:
                self._log_parse_failure(user_input, "llm_timeout", attempt)
                if attempt == self.max_retries:
                    raise RequirementParseError("LLM调用超时")

            except Exception as e:
                self._log_parse_failure(user_input, e, "llm", attempt)
                if attempt == self.max_retries:
                    raise RequirementParseError(f"LLM解析失败: {e}")

    def _validate_and_infer(self, raw_dict: dict) -> PPTRequirement:
        """
        Pydantic验证 + 智能推导

        Args:
            raw_dict: LLM输出的字典

        Returns:
            验证后的PPTRequirement模型
        """
        # 智能推导：填充默认值
        inferred = self._infer_defaults(raw_dict)

        # Pydantic验证
        try:
            return PPTRequirement(**inferred)
        except ValidationError as e:
            # 格式化错误提示
            error_msg = self._format_validation_error(e)
            logger.error(f"[{self.agent_name}] 验证失败: {error_msg}")
            raise RequirementParseError(error_msg) from e

    def _infer_defaults(self, raw_dict: dict) -> dict:
        """
        智能推导：根据已有字段推导缺失字段

        推导规则：
        1. scene → template_type
        2. template_type → tone
        3. 主题 → core_modules
        4. 全字段默认值
        """
        inferred = raw_dict.copy()

        # scene → template_type
        if 'scene' in inferred and 'template_type' not in inferred:
            inferred['template_type'] = self._infer_template_from_scene(inferred['scene'])

        # template_type → tone
        if 'template_type' in inferred and 'tone' not in inferred:
            inferred['tone'] = self._infer_tone_from_template(inferred['template_type'])

        # 主题 → core_modules
        if 'core_modules' not in inferred or not inferred['core_modules']:
            inferred['core_modules'] = self._extract_modules_from_topic(inferred.get('ppt_topic', ''))

        # 确保所有字段都有默认值
        defaults = {
            'page_num': 10,
            'language': 'ZH-CN',
            'template_type': 'business_template',
            'scene': 'business_report',
            'tone': 'professional',
            'core_modules': [],
            'need_research': False
        }

        for key, value in defaults.items():
            if key not in inferred or not inferred[key]:
                inferred[key] = value

        return inferred

    async def _parse_with_fallback(self, user_input: str) -> PPTRequirement:
        """
        降级解析（规则提取）

        保证：永远返回一个可用的PPTRequirement
        """
        logger.warning(f"[{self.agent_name}] 使用降级解析")

        # 规则提取
        fallback_dict = self._fallback_parse(user_input)

        # Pydantic验证（确保降级结果也符合模型）
        try:
            return PPTRequirement(**fallback_dict)
        except ValidationError as e:
            # 如果降级结果也验证失败，使用最简默认值
            logger.error(f"[{self.agent_name}] 降级结果验证失败，使用默认值: {e}")
            return PPTRequirement(
                ppt_topic=user_input[:100] or "未命名主题",
                page_num=10,
                language='ZH-CN'
            )
```
