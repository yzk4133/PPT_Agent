```python
import asyncio
import inspect
import pytest
from unittest.mock import AsyncMock, patch

class TestAsyncBasics:
    """异步基础特性测试"""

    @pytest.mark.asyncio
    async def test_parse_returns_coroutine(self):
        """验证parse()返回协程对象"""
        agent = RequirementParserAgent()

        # 不加await时返回协程对象
        coro = agent.parse("测试输入")
        assert inspect.iscoroutine(coro)

        # await后得到结果
        result = await coro
        assert isinstance(result, PPTRequirement)

    @pytest.mark.asyncio
```
