```python
class StreamingPPTGenerator:
    """流式 PPT 生成器"""

    async def generate_with_streaming(self, user_input: str):
        """
        流式生成 PPT，实时推送进度
        """

        # 阶段 1: 生成框架（30 秒）
        framework = await self.framework_agent.run(user_input)
        yield {
            "stage": "framework",
            "progress": 20,
            "result": framework
        }

        # 阶段 2: 研究页面（并行，60 秒）
        async for research_result in self.research_agent.stream_research(framework):
            yield {
                "stage": "research",
                "progress": 50,
                "current_page": research_result["page_no"],
                "result": research_result
            }

        # 阶段 3: 生成内容（逐页，90 秒）
        async for content_result in self.content_agent.stream_generate(framework):
            yield {
                "stage": "content",
                "progress": 50 + (content_result["page_no"] / total_pages * 40),
                "current_page": content_result["page_no"],
                "result": content_result
            }

        yield {
            "stage": "complete",
            "progress": 100,
            "result": final_result
        }
```

```python
class PPTVersionManager:
    """PPT 版本管理器"""

    async def create_version(
        self,
        ppt_id: str,
        version_type: str  # "major", "minor", "patch"
    ):
        """
        创建新版本
        """

        current_version = await self.get_current_version(ppt_id)

        new_version = self.increment_version(
            current_version,
            version_type
        )

        # 保存完整副本
        await self.save_version_snapshot(
            ppt_id,
            new_version
        )

        return new_version

    async def compare_versions(
        self,
        ppt_id: str,
        version_1: str,
        version_2: str
    ):
        """
        对比两个版本

        Returns:
            差异报告
        """

        v1 = await self.get_version(ppt_id, version_1)
        v2 = await self.get_version(ppt_id, version_2)

        comparison = {
            "version_1": version_1,
            "version_2": version_2,
            "differences": [],
            "summary": ""
        }

        # 对比每一页
        for page in v1["pages"]:
            v2_page = self.find_page(v2, page["page_no"])

            if self.pages_different(page, v2_page):
                comparison["differences"].append({
                    "page_no": page["page_no"],
                    "change_type": self.get_change_type(page, v2_page),
                    "v1_content": page["content"][:100] + "...",
                    "v2_content": v2_page["content"][:100] + "..."
                })

        # 生成摘要
        comparison["summary"] = await self.llm.ainvoke(f"""
        对比两个 PPT 版本的差异：

        {len(comparison['differences'])} 个页面有变化

        请生成一个简洁的摘要说明主要变化。
        """)

        return comparison

    async def restore_version(
        self,
        ppt_id: str,
        target_version: str
    ):
        """
        回溯到指定版本
        """

        version_snapshot = await self.get_version_snapshot(
            ppt_id,
            target_version
        )

        # 恢复到该版本
        await self.restore_from_snapshot(
            ppt_id,
            version_snapshot
        )

        return {
            "ppt_id": ppt_id,
            "restored_version": target_version,
            "timestamp": datetime.now().isoformat()
        }
```

```python
class RealChartGenerator:

                    },

                        "name": data["series_name"],
                        "type": "bar",
                        "data": data["values"]
                    }]
                },
                "title": {
                    "text": chart_config["title"],
                    "left": "center"
                },
                "tooltip": {...},
                "legend": {...}
            }

        elif chart_type == "line":
            # 生成折线图
            chart_spec = await self.generate_line_chart(
                chart_config,
                research_data
            )

        elif chart_type == "pie":
            # 生成饼图
            chart_spec = await self.generate_pie_chart(
                chart_config,
                research_data
            )

        return chart_spec

    async def extract_chart_data(
        self,
        research_data: dict,
        data_type: str
    ):
        """
        从研究数据中提取图表数据

        如果研究数据中没有数值数据，使用 LLM 生成合理的数据
        """

        # 尝试从研究中提取
        extracted = self.try_extract_data(research_data)

        if extracted:
            return extracted

        # 如果没有，生成合理的示例数据

        研究结果：{research_data}

        图表类型：{data_type}

        要求：
        - 数据应该真实可信（即使是示例也要合理）
        - 包含明确的数值
        - 数据点数量适中（5-10 个）

        返回 JSON 格式的图表数据。
        """

        return await self.llm.ainvoke(data_generation_prompt)
```

```python
class InteractivePPTGenerator:

        """

        understanding = await self.understand_requirements(user_input)

        confirmation = await self.ask_user(
            f"""
我理解您的需求是：
- 主题：{understanding['topic']}
- 页数：{understanding['page_count']}
- 风格：{understanding['style']}

是否正确？（输入 y/n，或提出修改意见）
"""
        )

            understanding = await self.refine_understanding(
                understanding,
                correction
            )

        framework = await self.framework_agent.run(understanding)

        framework_preview = self.format_framework_preview(framework)

        user_feedback = await self.ask_user(
            f"""
框架已生成：

{framework_preview}

是否满意？
1. 直接继续（输入 c）
2. 修改某页（输入页码和修改要求）
3. 重新设计框架（输入 r）
"""
        )

        if user_feedback.startswith("r"):
            # 重新设计

                understanding,

        elif user_feedback[0].isdigit():

                user_feedback

                framework,

1. 继续使用（输入 c）
2. 重新生成（输入 r）
3. 修改要求（输入修改意见）
"""
                )

                if satisfied == "r":

                        framework[page_no],
                        research_data,
```

```python
        scores["completeness"] = await self.score_completeness(
            ppt_content,
            user_requirements
        )

        # 2. 准确性评分（0-100）
        scores["accuracy"] = await self.score_accuracy(
            ppt_content,
            user_requirements
        )

        # 3. 逻辑性评分（0-100）
        scores["logic"] = await self.score_logic(
            ppt_content
        )

        # 4. 美观性评分（0-100）
        scores["aesthetics"] = await self.score_aesthetics(
            ppt_content
        )

        # 5. 创新性评分（0-100）
        scores["creativity"] = await self.score_creativity(
            ppt_content
        )

        # 计算总分
        overall_score = (
            scores["completeness"] * 0.25 +
            scores["accuracy"] * 0.30 +
            scores["logic"] * 0.25 +
            scores["aesthetics"] * 0.10 +
            scores["creativity"] * 0.10
        )

        return {
            "overall_score": overall_score,
            "dimension_scores": scores,
            "grade": self.calculate_grade(overall_score)
        }

    async def score_completeness(self, ppt, requirements):
        """评分：完整性"""

        required_elements = [
            "has_cover",
            "has_directory",
            "has_summary",
            "page_count_match"
        ]

        score = 0
        for element in required_elements:
            if self.check_element(ppt, element, requirements):
                score += 25

        return score
```
