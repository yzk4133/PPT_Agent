## 一、什么是模板系统？

### 1.1 核心概念

**模板系统**是指：预定义常见场景的 PPT 框架和风格，用户可以直接选择或系统自动推荐，从而**快速生成高质量、结构化的 PPT**。

```
传统方式：
用户需求 → 从零设计框架 → 生成内容 → 渲染
(每个任务都需要重新设计框架)

模板系统：
用户需求 → 选择/推荐模板 → 基于模板框架 → 生成内容 → 渲染
(直接使用预制的最优框架)
```

### 1.2 为什么需要模板系统？

| 问题 | 传统方式 | 模板系统 |
|------|---------|---------|
| **设计效率** | 每次从零设计 30 秒 | 直接使用模板 2 秒 |
| **框架质量** | 依赖 LLM 发挥不稳定 | 预制最佳实践框架 |
| **风格一致性** | 每次风格可能不同 | 同类型风格统一 |
| **专业度** | 需要专业知识 | 模板由专家设计 |
| **用户体验** | 等待时间长 | 快速、可预期 |

### 1.3 模板的价值

```
价值 1: 速度提升
  设计框架：30 秒 → 2 秒 (93% 提升)
  整体生成：3 分钟 → 2 分钟 (33% 提升)

价值 2: 质量保障
  预制框架经过验证
  避免常见的设计错误
  符合行业最佳实践

价值 3: 用户满意度
  看到模板就有预期
  减少试错成本
  提升专业信任度
```

```python
from typing import Dict, List, Any, Optional
from pydantic import BaseModel
from enum import Enum


class TemplateStyle(BaseModel):
    """模板风格定义"""
    color_scheme: str          # 配色方案
    font_family: str            # 字体
    layout_style: str           # 布局风格

    class Config:
        json_schema_extra = {
            "example": {
                "color_scheme": "professional_blue",
                "font_family": "Microsoft YaHei",
                "layout_style": "formal",
                "visual_style": "minimalist",
                "tone": "professional"
            }
        }


class PageTemplate(BaseModel):
    """页面模板"""
    page_no: int
    page_type: str              # cover, directory, content, chart, summary
    title_placeholder: str       # 标题模板
    content_prompt: str          # 内容生成提示
    estimated_word_count: int
    required_elements: List[str]  # 必需元素
    optional_elements: List[str]  # 可选元素


class PPTTemplate(BaseModel):
    """完整的 PPT 模板"""

    # 基本信息
    template_id: str              # 模板唯一标识
    template_name: str            # 模板名称
    category: str                 # 分类（business, tech, education, etc.）
    description: str              # 描述
    tags: List[str]               # 标签
    language: str = "zh-CN"        # 语言

    # 框架定义
    total_pages: int
    framework: List[PageTemplate]

    # 风格定义
    style: TemplateStyle

    # Agent 提示词（可选，定制化 Prompt）
    custom_prompts: Dict[str, str] = {
        "framework_agent": "",    # 框架设计 Agent 的 Prompt
        "research_agent": "",     # 研究 Agent 的 Prompt
        "content_agent": "",      # 内容生成 Agent 的 Prompt
    }

    # 适用条件
   适用场景: List[str]           # 适用场景描述
    recommended_page_range: Dict[str, int]  # 推荐页数范围
    required_inputs: List[str]   # 必需的用户输入

    # 元数据
    created_at: str
    updated_at: str
    version: str = "1.0"
    usage_count: int = 0         # 使用次数
    rating: float = 0.0          # 用户评分


class TemplateLibrary:
    """模板库管理"""

    def __init__(self):
        self.templates: Dict[str, PPTTemplate] = {}
        self.categories = set()
        self.load_templates()

    def load_templates(self):
        """加载所有模板"""
        # 可以从数据库、文件、API 加载
        self._load_builtin_templates()
        self._load_custom_templates()

    def _load_builtin_templates(self):
        """加载内置模板"""
        builtin_templates = [
            BusinessPlanTemplate(),
            TechShareTemplate(),
            EducationTemplate(),
            ReportTemplate(),
            CreativeTemplate()
        ]

        for template in builtin_templates:
            self.register_template(template)

    def get_template(self, template_id: str) -> Optional[PPTTemplate]:
        """获取模板"""
        return self.templates.get(template_id)

    def search_templates(
        self,
        category: Optional[str] = None,
        tags: Optional[List[str]] = None,
        page_count: Optional[int] = None
    ) -> List[PPTTemplate]:
        """搜索模板"""
        results = list(self.templates.values())

        if category:
            results = [t for t in results if t.category == category]

        if tags:
            results = [
                t for t in results
                if any(tag in t.tags for tag in tags)
            ]

        if page_count:
            min_pages = page_count - 3
            max_pages = page_count + 3
            results = [
                t for t in results
                if min_pages <= t.total_pages <= max_pages
            ]

        return results
```

### 3.2 动态模板生成

```python
class DynamicTemplateGenerator:
    """动态模板生成器"""

    async def generate_template_from_requirements(
        self,
        user_requirements: dict
    ) -> PPTTemplate:
        """
        根据用户需求动态生成模板

        适用场景：没有合适的内置模板时
        """

        requirements_prompt = f"""
        根据以下需求，生成一个 PPT 模板框架：

        用户需求：
        - 主题：{user_requirements['topic']}
        - 页数：{user_requirements['page_count']}
        - 用途：{user_requirements['purpose']}
        - 受众：{user_requirements['audience']}
        - 风格：{user_requirements['style']}

        请生成包含以下内容的模板：
        1. template_id: 唯一标识
        2. template_name: 模板名称
        3. framework: 详细的页面框架（{page_count} 页）
        4. style: 风格定义
        5. custom_prompts: 定制的 Agent Prompt

        以 JSON 格式返回完整模板。
        """

        template_dict = await self.llm.with_structured_output(
            PPTTemplate
        ).ainvoke(requirements_prompt)

        return PPTTemplate(**template_dict)


# 使用示例
generator = DynamicTemplateGenerator()

# 用户特殊需求

    "purpose": "技术分享",
    "audience": "技术人员",
    "style": "现代简约"
})
```

```python
class TemplateRecommendationAgent:
    """模板推荐 Agent"""

    def __init__(self, template_library: TemplateLibrary):
        self.template_library = template_library

    async def recommend(

        requirements: dict,

        """
        推荐最合适的模板

        Returns:
            推荐列表，按匹配度排序
        """

        # 1. 分析用户需求
        analysis = await self.analyze_user_requirements(
            user_input,
            requirements
        )

        # 2. 搜索候选模板
        candidates = self.search_candidates(
            analysis
        )

        # 3. 评分排序
        scored_templates = []
        for candidate in candidates:
            score = await self.score_template_match(
                candidate,
                analysis
            )

            scored_templates.append({
                "template": candidate,
                "score": score,
                "match_reasons": score["reasons"]
            })

        # 4. 排序并返回 top-k
        scored_templates.sort(key=lambda x: x["score"], reverse=True)

        return scored_templates[:top_k]

    async def analyze_user_requirements(
        self,
        user_input: str,
        requirements: dict
    ) -> Dict[str, Any]:
        """
        分析用户需求

        Returns:
            需求分析结果
        """

        analysis_prompt = f"""
        分析以下 PPT 生成需求：

        用户输入：{user_input}
        需求：{requirements}

        请分析并返回：
        {{
            "detected_category": "检测到的类别 (business|tech|education|creative|report)",
            "detected_purpose": "检测到的目的 (融资|分享|培训|报告|展示)",
            "detected_audience": "检测到的受众 (投资人|技术人员|学员|大众|客户)",
            "estimated_page_count": 预估页数,
            "tone_preference": "语气偏好 (正式|轻松|专业|友好)",
            "style_preference": "风格偏好 (简约|华丽|科技|商务)",
            "keywords": ["关键词1", "关键词2"]
        }}
        """

        return await self.llm.with_structured_output(
            RequirementAnalysis
        ).ainvoke(analysis_prompt)

    def search_candidates(
        self,
        analysis: Dict[str, Any]
    ) -> List[PPTTemplate]:
        """搜索候选模板"""

        candidates = []

        # 按类别筛选
        category_templates = self.template_library.search_templates(
            category=analysis["detected_category"]
        )

        # 按页数筛选
        page_range = analysis.get("estimated_page_count", 10)
        matching_templates = self.template_library.search_templates(
            page_count=page_range
        )

        # 合并并去重
        all_templates = list(set(
            category_templates + matching_templates
        ))

        return all_templates

    async def score_template_match(
        self,
        template: PPTTemplate,
        analysis: Dict[str, Any]
    ) -> Dict[str, Any]:
        """
        评分模板匹配度

        评分维度：
        1. 类别匹配 (0-1)
        2. 页数匹配 (0-1)
        3. 标签匹配 (0-1)
        4. 风格匹配 (0-1)
        """

        scores = {}

        # 1. 类别匹配
        scores["category"] = 1.0 if \
            template.category == analysis["detected_category"] else 0.5

        # 2. 页数匹配
        expected_pages = analysis.get("estimated_page_count", 10)
        page_diff = abs(template.total_pages - expected_pages)
        scores["page_count"] = max(0, 1 - page_diff / 10)

        # 3. 标签匹配
        user_keywords = set(analysis.get("keywords", []))
        template_tags = set(template.tags)
        if user_keywords and template_tags:
            intersection = user_keywords & template_tags
            scores["tags"] = len(intersection) / len(user_keywords)
        else:
            scores["tags"] = 0.5

        # 4. 风格匹配
        style_keywords = {
            "professional": ["business", "formal", "serious"],
            "casual": ["creative", "friendly", "engaging"],
            "technical": ["tech", "minimalist", "modern"],
            "friendly": ["education", "encouraging", "warm"]
        }

        user_style = analysis.get("style_preference", "professional")
        if template.style.tone.lower() in style_keywords.get(user_style, []):
            scores["style"] = 1.0
        else:
            scores["style"] = 0.5

        # 计算加权总分
        weights = {"category": 0.3, "page_count": 0.2, "tags": 0.3, "style": 0.2}

        overall_score = (
            scores["category"] * weights["category"] +
            scores["page_count"] * weights["page_count"] +
            scores["tags"] * weights["tags"] +
            scores["style"] * weights["style"]
        )

        # 生成匹配原因
        reasons = []
        if scores["category"] == 1.0:
            reasons.append(f"类别完全匹配（{template.category}）")
        if scores["page_count"] > 0.8:
            reasons.append(f"页数合适（{template.total_pages}页）")
        if scores["tags"] > 0.5:
            reasons.append(f"标签匹配度高")

        return {
            "overall_score": overall_score,
            "dimension_scores": scores,
            "reasons": reasons
        }
```

## 五、模板应用流程

### 5.1 完整流程

```python
class TemplateBasedPPTGenerator:
    """基于模板的 PPT 生成器"""

    def __init__(
        self,
        template_library: TemplateLibrary,
        template_recommender: TemplateRecommendationAgent,
        master_graph: MasterGraph
    ):
        self.template_library = template_library
        self.recommender = template_recommender
        self.master_graph = master_graph

    async def generate_with_template(
        self,
        user_input: str,
        requirements: dict
    ):
        """
        基于模板生成 PPT 的完整流程
        """

        # 阶段 1: 推荐模板
        logger.info(f"[TemplateGenerator] 分析需求并推荐模板...")

        recommendations = await self.recommender.recommend(
            user_input,
            requirements,
            top_k=3
        )

        # 展示推荐（或返回给用户选择）
        selected_template = await self.present_and_select_template(
            recommendations,
            user_input
        )

        # 阶段 2: 应用模板
        logger.info(f"[TemplateGenerator] 应用模板: {selected_template.template_name}")

        framework, agent_prompts, style = await self.apply_template(
            selected_template,
            user_input,
            requirements
        )

        # 阶段 3: 生成内容
        logger.info(f"[TemplateGenerator] 基于模板框架生成内容...")

        result = await self.generate_with_framework(
            framework,
            agent_prompts,
            style
        )

        return result

    async def present_and_select_template(
        self,
        recommendations: List[Dict],
        user_input: str
    ) -> PPTTemplate:
        """
        展示推荐并选择模板

        默认：选择最高分的模板
        可选：让用户选择
        """

        # 方案 A: 自动选择（默认）
        return recommendations[0]["template"]

        # 方案 B: 用户选择
        # 选择逻辑：
        # 1. 展示推荐给用户
        # 2. 用户输入选择（1/2/3）
        # 3. 或用户输入"custom"自定义

        # 简化版：返回最佳匹配
        best_match = recommendations[0]["template"]

        logger.info(
            f"[TemplateGenerator] 自动选择最佳匹配: "
            f"{best_match.template_name} "
            f"(匹配度: {recommendations[0]['score']:.1%})"
        )

        return best_match

    async def apply_template(
        self,
        template: PPTTemplate,
        user_input: str,
        requirements: dict
    ):
        """
        应用模板

        Returns:
            framework, agent_prompts, style
        """

        # 1. 使用模板的预设框架
        framework = {
            "total_page": template.total_pages,
            "ppt_framework": [
                {
                    "page_no": page.page_no,
                    "title": "",  # 将由后续填充
                    "page_type": page.page_type,
                    "content_prompt": page.content_prompt,
                    "estimated_word_count": page.estimated_word_count,
                    "required_elements": page.required_elements,
                    "optional_elements": page.optional_elements
                }
                for page in template.framework
            ],
            "framework_type": "template_based"
        }

        # 2. 如果模板有定制 Prompt，使用定制 Prompt
        agent_prompts = template.custom_prompts or {}

        # 3. 应用模板风格
        style = template.style

        return framework, agent_prompts, style

    async def generate_with_framework(
        self,
        framework: dict,
        agent_prompts: Dict[str, str],
        style: TemplateStyle
    ):
        """
        基于框架生成 PPT（使用自定义 Prompt）
        """

        # 创建定制的 Agent 实例
        if agent_prompts:
            research_agent = ResearchAgent(
                prompt=agent_prompts.get("research_agent")
            )
            content_agent = ContentMaterialAgent(
                prompt=agent_prompts.get("content_agent")
            )
        else:
            research_agent = self.recommender.template_library.default_agents["research"]
            content_agent = self.recommender.template_library.default_agents["content"]

        # 运行生成流程
        state = {
            "user_input": requirements.get("user_input"),
            "structured_requirements": requirements,
            "ppt_framework": framework,
            "style": style
        }

        # 执行工作流（研究 → 内容生成）
        result = await self.master_graph.invoke(state)

        return result
```

```python
# agents/coordinator/master_graph.py

class MasterGraph:
    def __init__(
        self,
        use_templates: bool = True,  # 新增参数
        template_library: Optional[TemplateLibrary] = None
    ):
        self.use_templates = use_templates
        self.template_library = template_library or TemplateLibrary()

    async def generate(self, user_input: str, requirements: dict):
        """主生成流程"""

        if self.use_templates:
            # 🔧 模板流程
            return await self.generate_with_template_flow(
                user_input,
                requirements
            )
        else:
            # 原流程（从零设计）
            return await self.generate_from_scratch(
                user_input,
                requirements
            )

    async def generate_with_template_flow(self, user_input, requirements):
        """基于模板的生成流程"""

        # 1. 推荐模板
        recommender = TemplateRecommendationAgent(self.template_library)
        recommendations = await recommender.recommend(user_input, requirements)

        # 2. 选择模板（自动或用户选择）
        template = recommendations[0]["template"]

        # 3. 应用模板
        framework = self.apply_template(template, requirements)

        # 4. 执行生成（研究 → 内容 → 渲染）
        result = await self.execute_with_framework(framework)

        return result
```
