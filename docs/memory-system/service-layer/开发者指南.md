## 为什么需要业务层？

### 问题1: 数据库操作复杂性

**❌ 没有业务层时：**
```python
class MyAgent:
    async def process(self, user_id: str):
        # 应用层需要处理：
        # 1. SQL查询
        # 2. 缓存逻辑
        # 3. 事务管理
        # 4. 错误处理
        result = self.db_session.query(UserProfile).filter(
            UserProfile.user_id == user_id
        ).first()
        if result:
            await self.cache.set(f"user:{user_id}", result.to_dict())
```

**✅ 有业务层时：**
```python
class MyAgent:
    async def process(self, user_id: str):
        # 简洁、清晰的业务调用
        preferences = await self._user_pref_service.get_user_preferences(user_id)
```

### 问题2: 代码重复

多个Agent都需要获取用户偏好，没有Service层会导致代码重复。

### 问题3: 数据一致性

更新数据后需要同步缓存，没有Service层容易忘记更新缓存。

**核心价值：** 封装复杂性、代码复用、数据一致性、易于测试、易于维护

## 设计原则

### 1. 单一职责原则

每个Service只负责一个业务领域。

```python
# ✅ 好的设计
class UserPreferenceService:
    """只处理用户偏好相关逻辑"""
    async def get_user_preferences(self, user_id): ...
    async def update_preferences(self, user_id, prefs): ...

# ❌ 不好的设计
class MemoryService:
    """处理所有记忆相关逻辑（职责太多）"""
    async def get_user_preferences(self, user_id): ...
    async def record_decision(self, decision): ...
    async def share_workspace_data(self, data): ...
```

### 2. 依赖注入原则

Service通过构造函数注入依赖，而不是硬编码。

```python
# ✅ 好的设计
class UserPreferenceService:
    def __init__(
        self,
        db_session: Optional[Session] = None,
        cache_client: Optional[RedisCache] = None,
        enable_cache: bool = True
    ):
        self.db_session = db_session
        self.cache_client = cache_client
        self.enable_cache = enable_cache

# 使用时注入
service = UserPreferenceService(
    db_session=session,
    cache_client=redis_client,
    enable_cache=True
)
```

### 3. 优雅降级原则

当存储不可用时，Service应该返回默认值，而不是崩溃。

```python
# ✅ 好的设计
async def get_user_preferences(self, user_id: str):
    if self.enable_cache:
        cached = await self.cache_client.get_user_preferences(user_id)
        if cached:
            return cached

    if self.db_session:
        try:
            profile = self._get_profile(user_id)
            return profile.to_dict()
        except Exception as e:
            self.logger.error(f"Database error: {e}")

    return {}  # 返回默认值，不崩溃

# ❌ 不好的设计
async def get_user_preferences(self, user_id: str):
    if not self.db_session:
        raise Exception("Database not available")  # Agent会崩溃
```

### 4. 防御性编程原则

Service应该验证参数、处理异常、回滚事务。

## 编码规范

### 命名规范

```python
# ✅ 好的命名（动词开头，清晰表达意图）
async def get_user_preferences(user_id):     # 获取
async def update_preferences(user_id, prefs): # 更新
async def delete_preferences(user_id):        # 删除
async def increment_session_count(user_id):   # 增加

# ❌ 不好的命名
async def preferences(user_id):        # 不明确是获取还是设置
async def do_update(user_id, data):    # do是多余的
async def processData(data):            # 驼峰命名不符合Python规范
```

### 返回值规范

```python
# ✅ 好的设计：返回类型一致
async def get_user_preferences(user_id) -> Dict[str, Any]:
    """总是返回字典（成功有数据，失败返回空字典）"""
    try:
        return profile.to_dict()
    except:
        return {}  # 一致的返回类型

async def update_preferences(user_id, prefs) -> bool:
    """总是返回布尔值（成功True，失败False）"""
    try:
        # 更新逻辑
        return True
    except:
        return False
```

### 日志规范

```python
# ✅ 好的日志（分级清晰）
self.logger.debug(f"Cache hit for user {user_id}")    # 调试信息
self.logger.info(f"Updated preferences for {user_id}")  # 重要操作
self.logger.warning(f"Database not available")        # 警告
self.logger.error(f"Failed to update: {e}")           # 错误

# ❌ 不好的日志
print("Updated preferences")  # 不要用print
self.logger.info(f"User {user_id} with prefs {prefs} updated...")  # 太详细，用debug
```

## 如何编写Service

### 步骤1: 创建Service类

```python
from typing import Optional, Dict, Any
from sqlalchemy.orm import Session
import logging
from ..storage.redis_cache import RedisCache

class MyService:
    """
    我的服务类

    职责：[描述这个Service的职责]
    """
    def __init__(
        self,
        db_session: Optional[Session] = None,
        cache_client: Optional[RedisCache] = None,
        enable_cache: bool = True
    ):
        """
        初始化服务

        Args:
            db_session: 数据库会话
            cache_client: Redis缓存客户端
            enable_cache: 是否启用缓存
        """
        self.db_session = db_session
        self.cache_client = cache_client
        self.logger = logging.getLogger(self.__class__.__name__)
        self.enable_cache = enable_cache and cache_client is not None
```

### 步骤2: 实现查询方法

```python
    async def get_data(self, key: str) -> Optional[Dict[str, Any]]:
        """
        获取数据

        Args:
            key: 数据键

        Returns:
            数据字典，未找到返回None
        """
        # 1. 优雅降级
        if not self.db_session:
            self.logger.warning("Database session not available")
            return None

        # 2. 先查缓存
        if self.enable_cache:
            try:
                cached = await self.cache_client.get(f"my_data:{key}")
                if cached:
                    self.logger.debug(f"Cache hit for key: {key}")
                    return cached
            except Exception as e:
                self.logger.warning(f"Cache error: {e}")

        # 3. 查数据库
        try:
            data = self.db_session.query(MyModel).filter(
                MyModel.key == key
            ).first()

            if data:
                # 4. 更新缓存
                if self.enable_cache:
                    try:
                        await self.cache_client.set(
                            f"my_data:{key}",
                            data.to_dict(),
                            ttl=3600
                        )
                    except Exception as e:
                        self.logger.warning(f"Cache set error: {e}")
                return data.to_dict()

            return None

        except Exception as e:
            self.logger.error(f"Database error: {e}")
            return None
```

### 步骤3: 实现更新方法

```python
    async def update_data(self, key: str, value: Any) -> bool:
        """
        更新数据

        Args:
            key: 数据键
            value: 新值

        Returns:
            是否成功
        """
        # 1. 优雅降级
        if not self.db_session:
            return False

        # 2. 参数验证
        if not key:
            self.logger.warning("Key is empty")
            return False

        try:
            # 3. 查询现有数据
            data = self.db_session.query(MyModel).filter(
                MyModel.key == key
            ).first()

            if not data:
                data = MyModel(key=key, value=value)
                self.db_session.add(data)
            else:
                data.value = value

            # 4. 提交事务
            self.db_session.commit()

            # 5. 使缓存失效
            if self.enable_cache:
                try:
                    await self.cache_client.delete(f"my_data:{key}")
                except Exception as e:
                    self.logger.warning(f"Cache invalidation error: {e}")

            self.logger.info(f"Updated data: {key}")
            return True

        except Exception as e:
            # 6. 错误处理
            self.db_session.rollback()
            self.logger.error(f"Update error: {e}")
            return False
```

## 事务与缓存管理

### 标准事务模式

```python
# ✅ 好的事务管理
async def update_data(self, key, value):
    try:
        # 业务逻辑
        data = self.db_session.query(Model).filter(...).first()
        data.value = value

        # 提交事务
        self.db_session.commit()

        # 使缓存失效
        await self.cache_client.delete(f"cache:{key}")

        return True

    except Exception as e:
        # 失败时回滚
        self.db_session.rollback()
        self.logger.error(f"Update failed: {e}")
        return False
```

### 缓存一致性

**关键原则：更新数据库后立即使缓存失效**

```python
# ✅ 好的缓存管理
async def update_data(self, key, value):
    # 1. 更新数据库
    data.value = value
    self.db_session.commit()

    # 2. 使缓存失效（保证一致性）
    await self.cache_client.delete(f"cache:{key}")

    return True

# ❌ 不好的缓存管理
async def update_data(self, key, value):
    data.value = value
    self.db_session.commit()
    # 忘记更新缓存！下次读取会得到旧数据
    return True
```

### 批量操作事务

```python
# ✅ 批量操作使用同一个事务
async def update_batch(self, updates: List[Dict]):
    try:
        for update in updates:
            data = self.db_session.query(Model).get(update["id"])
            data.value = update["value"]
            # 不立即commit

        self.db_session.commit()  # 一次性提交所有更改
        return True

    except Exception as e:
        self.db_session.rollback()  # 失败时全部回滚
        self.logger.error(f"Batch update failed: {e}")
        return False
```

## 📊 快速检查清单

编写Service时检查：

- [ ] 类名清晰表达职责（如UserPreferenceService）
- [ ] 方法名动词开头（get/update/delete）
- [ ] 所有方法有类型提示
- [ ] 所有方法有docstring
- [ ] 返回值类型一致
- [ ] 异常被捕获并记录日志
- [ ] 事务正确管理（commit/rollback）
- [ ] 缓存一致性保证（更新后失效）
- [ ] 使用合适的日志级别
- [ ] 优雅降级（依赖不可用时返回默认值）
- [ ] 参数验证（检查None、空字符串等）
