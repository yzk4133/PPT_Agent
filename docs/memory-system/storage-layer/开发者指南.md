## 为什么需要存储层？

### 问题1: 数据分散存储

**❌ 没有存储层时：**
```python
# 数据分散在各个地方
class MyAgent:
    def __init__(self):
        # 用户偏好存在文件
        self.user_prefs = self.load_from_file("prefs.json")

        # 会话数据存在内存
        self.sessions = {}

        # 对话历史存在数据库
        self.db = sqlite3.connect("chat.db")

    # 需要处理3种不同的存储方式
    def get_data(self, key):
        if key in self.user_prefs:
            return self.user_prefs[key]
        elif key in self.sessions:
            return self.sessions[key]
        else:
            return self.query_db(key)
```

**✅ 有存储层时：**
```python
# 统一的数据访问接口
from backend.memory.storage import get_db_session, RedisCache

class MyAgent:
    def __init__(self):
        self.db_session = get_db_session()  # 数据库
        self.cache = RedisCache()            # 缓存

    def get_data(self, key):
        # 统一的访问方式
        return self.db_session.query(Model).filter_by(key=key).first()
```

### 问题2: 连接管理复杂

**❌ 手动管理连接：**
```python
# 容易忘记关闭连接
conn = mysql.connect(host, user, password)
cursor = conn.cursor()
cursor.execute("SELECT * FROM users")
# 忘记关闭连接！
```

**✅ 使用连接池：**
```python
# 存储层自动管理连接池
db = get_db()
with db.get_session() as session:
    users = session.query(User).all()
# 自动关闭连接
```

### 问题3: 性能优化困难

**❌ 每次都查数据库：**
```python
# 每次都查数据库，性能差
def get_user_preferences(user_id):
    profile = db.query(UserProfile).filter_by(user_id=user_id).first()
    return profile.to_dict()
```

**✅ 使用缓存：**
```python
# 先查缓存，提升性能
async def get_user_preferences(user_id):
    # 1. 查缓存（1-5ms）
    cached = await cache.get_user_preferences(user_id)
    if cached:
        return cached

    # 2. 查数据库（10-50ms）
    profile = db.query(UserProfile).filter_by(user_id=user_id).first()

    # 3. 写缓存
    await cache.set_user_preferences(user_id, profile.to_dict())
    return profile.to_dict()
```

### 核心价值

| 价值 | 说明 |
|------|------|
| **统一接口** | 数据库、缓存、文件统一访问 |
| **连接管理** | 连接池、自动重连、健康检查 |
| **性能优化** | Redis缓存减少数据库压力 |
| **数据一致性** | 统一的事务管理 |
| **易于测试** | 可以轻松Mock存储层 |

## 存储层设计原则

### 1. 单一职责原则

每个组件只负责一件事：

```python
# ✅ 好的设计
class DatabaseManager:
    """只负责数据库连接管理"""
    pass

class RedisCache:
    """只负责缓存管理"""
    pass

class UserProfile(BaseModel):
    """只负责用户配置数据模型"""
    pass

# ❌ 不好的设计
class StorageManager:
    """既管数据库又管缓存又管模型（职责太多）"""
    pass
```

### 2. 连接池原则

使用连接池，避免频繁创建连接：

```python
# ✅ 好的设计 - 使用连接池
engine = create_engine(
    DATABASE_URL,
    poolclass=QueuePool,  # 连接池
    pool_size=10,         # 池大小
    max_overflow=20       # 最大溢出
)

# ❌ 不好的设计 - 每次创建新连接
def query_data():
    conn = mysql.connect(...)  # 每次都创建新连接
    ...
    conn.close()
```

### 3. 缓存优先原则

先查缓存，再查数据库：

```python
# ✅ 好的设计
async def get_data(key):
    # 1. 查缓存（快）
    cached = await cache.get(key)
    if cached:
        return cached

    # 2. 查数据库（慢）
    data = db.query(Model).filter_by(key=key).first()

    # 3. 写缓存
    await cache.set(key, data)
    return data
```

### 4. 优雅降级原则

缓存不可用时，不应该影响主流程：

```python
# ✅ 好的设计 - 缓存失败不影响数据库
async def get_data(key):
    # 尝试缓存
    try:
        cached = await cache.get(key)
        if cached:
            return cached
    except Exception as e:
        logger.warning(f"Cache error: {e}")

    # 降级到数据库
    return db.query(Model).filter_by(key=key).first()
```

## 数据库编程

### 创建数据库管理器

```python
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker, Session
from typing import Optional
import logging

class DatabaseManager:
    """数据库管理器 - 单例模式"""

    _instance: Optional["DatabaseManager"] = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialized = False
        return cls._instance

    def __init__(self):
        if self._initialized:
            return

        # 1. 创建引擎
        self.engine = create_engine(
            DATABASE_URL,
            poolclass=QueuePool,
            pool_size=10,
            max_overflow=20,
            pool_pre_ping=True  # 连接前检查
        )

        # 2. 创建会话工厂
        self.SessionLocal = sessionmaker(
            autocommit=False,
            autoflush=False,
            bind=self.engine
        )

        self._initialized = True
        logger.info("DatabaseManager initialized")

    def get_session(self) -> Session:
        """获取会话"""
        return self.SessionLocal()

    def health_check(self) -> bool:
        """健康检查"""
        try:
            with self.engine.connect() as conn:
                conn.execute(text("SELECT 1"))
            return True
        except Exception as e:
            logger.error(f"Health check failed: {e}")
            return False
```

### 使用会话（上下文管理器）

```python
from contextlib import contextmanager

@contextmanager
def get_session(self):
    """上下文管理器 - 自动commit/rollback"""
    session = self.SessionLocal()
    try:
        yield session
        session.commit()
    except Exception as e:
        session.rollback()
        logger.error(f"Session error: {e}")
        raise
    finally:
        session.close()

# 使用
with db.get_session() as session:
    user = session.query(User).first()
    user.name = "New Name"
    # 自动提交
```

### 查询数据

```python
# 1. 简单查询
user = session.query(User).filter_by(user_id="user123").first()

# 2. 条件查询
users = session.query(User).filter(
    User.age >= 18,
    User.status == "active"
).all()

# 3. 排序
users = session.query(User).order_by(User.created_at.desc()).limit(10).all()

# 4. 聚合
count = session.query(User).count()
```

### 更新数据

```python
# 1. 先查询再更新
user = session.query(User).filter_by(user_id="user123").first()
if user:
    user.name = "New Name"
    session.commit()

# 2. 批量更新
session.query(User).filter(
    User.status == "inactive"
).update({"active": False}, synchronize_session=False)
session.commit()
```

### 删除数据

```python
# 1. 先查询再删除
user = session.query(User).filter_by(user_id="user123").first()
if user:
    session.delete(user)
    session.commit()

# 2. 批量删除
session.query(User).filter(
    User.created_at < datetime(2020, 1, 1)
).delete(synchronize_session=False)
session.commit()
```

## 缓存编程

### 创建Redis缓存管理器

```python
from redis import Redis
from redis.exceptions import RedisError
from typing import Optional, Dict, Any
import json
import logging

class RedisCache:
    """Redis缓存管理器"""

    def __init__(self, redis_url: str):
        try:
            self.client = Redis.from_url(
                redis_url,
                decode_responses=True,  # 自动解码
                socket_timeout=5,
                retry_on_timeout=True,
                health_check_interval=30
            )

            # 测试连接
            self.client.ping()
            self.available = True
            logger.info("Redis connected")

        except RedisError as e:
            logger.warning(f"Redis connection failed: {e}")
            self.client = None
            self.available = False

    def is_available(self) -> bool:
        """检查是否可用"""
        if not self.client:
            return False
        try:
            self.client.ping()
            return True
        except RedisError:
            return False
```

### 基础缓存操作

```python
# 1. 设置缓存
async def set(self, key: str, value: str, ttl: int = 3600) -> bool:
    """设置缓存（带TTL）"""
    if not self.client:
        return False

    try:
        self.client.setex(key, ttl, value)
        return True
    except RedisError as e:
        logger.warning(f"Set error: {e}")
        return False

# 2. 获取缓存
async def get(self, key: str) -> Optional[str]:
    """获取缓存"""
    if not self.client:
        return None

    try:
        return self.client.get(key)
    except RedisError as e:
        logger.warning(f"Get error: {e}")
        return None

# 3. 删除缓存
async def delete(self, key: str) -> bool:
    """删除缓存"""
    if not self.client:
        return False

    try:
        return self.client.delete(key) > 0
    except RedisError as e:
        logger.warning(f"Delete error: {e}")
        return False
```

### JSON缓存

```python
# 1. 设置JSON
async def set_json(self, key: str, value: Dict, ttl: int = 3600) -> bool:
    """设置JSON缓存"""
    try:
        json_str = json.dumps(value, ensure_ascii=False)
        return await self.set(key, json_str, ttl)
    except (TypeError, ValueError) as e:
        logger.warning(f"JSON encode error: {e}")
        return False

# 2. 获取JSON
async def get_json(self, key: str) -> Optional[Dict]:
    """获取JSON缓存"""
    value = await self.get(key)
    if value:
        try:
            return json.loads(value)
        except json.JSONDecodeError as e:
            logger.warning(f"JSON decode error: {e}")
    return None
```

### 业务特定缓存方法

```python
# 用户偏好缓存
async def get_user_preferences(self, user_id: str) -> Optional[Dict]:
    """获取用户偏好缓存"""
    key = f"user_pref:{user_id}"
    return await self.get_json(key)

async def set_user_preferences(
    self, user_id: str, preferences: Dict
) -> bool:
    """设置用户偏好缓存"""
    key = f"user_pref:{user_id}"
    return await self.set_json(key, preferences, ttl=86400)  # 24小时

# 会话缓存
async def get_session(self, session_id: str) -> Optional[Dict]:
    """获取会话缓存"""
    key = f"session:{session_id}"
    return await self.get_json(key)

async def set_session(self, session_id: str, data: Dict) -> bool:
    """设置会话缓存"""
    key = f"session:{session_id}"
    return await self.set_json(key, data, ttl=3600)  # 1小时
```

### 批量操作

```python
async def invalidate_pattern(self, pattern: str) -> int:
    """批量删除匹配模式的缓存"""
    if not self.client:
        return 0

    try:
        keys = self.client.keys(pattern)
        if keys:
            deleted = self.client.delete(*keys)
            logger.info(f"Invalidated {deleted} keys: {pattern}")
            return deleted
        return 0
    except RedisError as e:
        logger.warning(f"Pattern delete error: {e}")
        return 0

# 使用
await cache.invalidate_pattern("user_pref:*")  # 删除所有用户偏好缓存
```

## 数据模型设计

### 创建基类

```python
from sqlalchemy import Column, DateTime
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime
from typing import Dict, Any

Base = declarative_base()

class BaseModel(Base):
    """模型基类"""
    __abstract__ = True

    # 通用时间戳
    created_at = Column(DateTime, default=datetime.utcnow, nullable=False)
    updated_at = Column(
        DateTime,
        default=datetime.utcnow,
        onupdate=datetime.utcnow,
        nullable=False
    )

    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        raise NotImplementedError()
```

### 创建具体模型

```python
from sqlalchemy import Column, String, Integer, Text
import uuid

class UserProfile(BaseModel):
    """用户配置表"""
    __tablename__ = "user_profiles"

    # 主键
    id = Column(String(36), primary_key=True, default=lambda: str(uuid.uuid4()))
    user_id = Column(String(255), unique=True, nullable=False, index=True)

    # 用户偏好（JSON格式）
    preferences = Column(Text, default="{}")

    # 使用统计
    total_sessions = Column(Integer, default=0)
    successful_generations = Column(Integer, default=0)

    # 满意度评分
    avg_satisfaction_score = Column(Integer, default=0)

    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        import json
        preferences = json.loads(self.preferences) if self.preferences else {}

        return {
            "id": str(self.id),
            "user_id": self.user_id,
            "preferences": preferences,
            "total_sessions": self.total_sessions,
            "successful_generations": self.successful_generations,
            "avg_satisfaction_score": self.avg_satisfaction_score,
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat(),
        }

    def get_preference(self, key: str, default=None) -> Any:
        """获取单个偏好"""
        import json
        prefs = json.loads(self.preferences) if self.preferences else {}
        return prefs.get(key, default)

    def set_preference(self, key: str, value: Any) -> None:
        """设置单个偏好"""
        import json
        prefs = json.loads(self.preferences) if self.preferences else {}
        prefs[key] = value
        self.preferences = json.dumps(prefs, ensure_ascii=False)
```

### 模型设计最佳实践

```python
# ✅ 好的设计
class UserProfile(BaseModel):
    """用户配置表"""
    __tablename__ = "user_profiles"

    # 1. 使用UUID作为主键
    id = Column(String(36), primary_key=True, default=lambda: str(uuid.uuid4()))

    # 2. 业务字段加索引
    user_id = Column(String(255), unique=True, nullable=False, index=True)

    # 3. JSON字段用Text存储（兼容MySQL）
    preferences = Column(Text, default="{}")

    # 4. 提供辅助方法
    def get_preference(self, key: str):
        ...

    def set_preference(self, key: str, value: Any):
        ...

    def to_dict(self) -> Dict:
        ...

# ❌ 不好的设计
class UserProfile(BaseModel):
    """用户配置表"""
    __tablename__ = "user_profiles"

    # 1. 使用自增ID（不利于分布式）
    id = Column(Integer, primary_key=True, autoincrement=True)

    # 2. 业务字段没有索引（查询慢）
    user_id = Column(String(255))

    # 3. 直接用JSON类型（MySQL兼容性差）
    preferences = Column(JSON)

    # 4. 没有辅助方法（使用不便）
```
