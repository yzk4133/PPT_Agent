## 项目概述

### 目标

构建一个智能客服意图识别系统，将用户自然语言输入转换为结构化 JSON 输出。

### 示例

```json
输入: "帮我查一下明天北京到上海的机票"
输出: {
  "intent": "query_flight",
  "confidence": 0.95,
  "slots": {
    "departure": "北京",
    "destination": "上海",
    "date": "明天"
  }
}
```

### 意图类型

```
1. query_flight    - 查询航班
2. book_flight    - 预订航班
3. query_weather  - 查询天气
4. set_alarm     - 设置闹钟
5. play_music    - 播放音乐
6. general_chat   - 通用对话
```

```python
import json

# 定义意图模板
templates = {
    "query_flight": [
        "帮我查{date}{origin}到{destination}的机票",

        "查询{date}从{origin}出发到{destination}的飞机",
    ],
    "book_flight": [
        "我要订{date}{origin}到{destination}的机票",

    ],
    "query_weather": [
        "查一下{date}{location}的天气",
        "{location}明天会下雨吗",
        "看看{date}{location}天气怎么样",
        "告诉我{location}{date}的气温",
    ],
    "set_alarm": [
        "设置一个{time}的闹钟",
        "在{time}提醒我",
        "{time}叫醒我",
        "帮我定个闹钟，时间{time}",
    ],
    "play_music": [
        "放一首{song}",
        "听{artist}的歌",
        "播放{genre}音乐",
        "来点{song}",
    ],
    "general_chat": [
        "你好",
        "请介绍一下自己",
        "你能做什么",
        "谢谢",
    ],
}

# 填充槽位
def generate_training_data():
    """生成训练数据"""
    data = []

    # 槽位值库
    slots = {
        "origin": ["北京", "上海", "深圳", "杭州", "成都"],
        "destination": ["上海", "北京", "深圳", "杭州", "成都"],
        "date": ["明天", "后天", "下周一", "本周五", "大后天"],
        "passengers": ["一张", "两张", "三张", "1张", "2张"],
        "location": ["北京", "上海", "深圳", "杭州", "广州"],
        "time": ["早上7点", "8点", "晚上9点", "明天早上", "半小时后"],
        "song": ["《青花瓷》", "《夜曲》", "流行歌曲", "轻音乐"],
        "artist": ["周杰伦", "邓紫棋", "陈奕迅"],
        "genre": ["流行", "摇滚", "古典", "民谣"],
    }

    # 生成数据
    for intent, patterns in templates.items():
        for pattern in patterns:
            # 尝试填充模板
            try:
                text = pattern.format(**slots)
            except KeyError:
                continue

            # 构造对应的输出
            if intent == "general_chat":
                output = {"intent": intent, "slots": {}}
            elif intent == "query_flight":
                output = {
                    "intent": intent,
                    "slots": {
                        "departure": slots["origin"][0],
                        "destination": slots["destination"][0],
                        "date": slots["date"][0]
                    }
                }
            elif intent == "query_weather":
                output = {
                    "intent": intent,
                    "slots": {
                        "location": slots["location"][0],
                        "date": slots["date"][0]
                    }
                }
            # ... 其他意图类似

            # 转换为训练格式
            data.append({
                "conversations": [
                    {"role": "user", "content": text},
                    {"role": "assistant", "content": json.dumps(output, ensure_ascii=False)}
                ]
            })

    # 保存
    with open('dataset/lora_intent.jsonl', 'w', encoding='utf-8') as f:
        for item in data:
            f.write(json.dumps(item, ensure_ascii=False) + '\n')

    print(f"生成了 {len(data)} 条训练数据")

generate_training_data()
```

#### 方式 B: 使用大模型生成（快速但需验证）

```python
import json
from openai import OpenAI

client = OpenAI(api_key="your-api-key", base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")

def generate_intent_data(intent, description, examples, count=50):
    """使用大模型生成训练数据"""

    prompt = f"""
    你是一个专业的训练数据生成助手。

    意图类型: {intent}
    描述: {description}

    参考示例:
    {json.dumps(examples, ensure_ascii=False, indent=2)}

    请生成 {count} 条不同的用户输入，要求：
    1. 表达方式多样化（口语、书面、省略主语等）
    2. 包含不同的槽位值
    3. 语法自然但不刻意
    4. 返回JSON数组格式

    每条数据包含：
    - input: 用户输入文本
    - output: 对应的JSON输出（intent + slots）
    """

    response = client.chat.completions.create(
        model="qwen-plus",
        messages=[{"role": "user", "content": prompt}],
        temperature=0.8
    )

    # 解析并验证
    try:
        generated = json.loads(response.choices[0].message.content)

        # 转换为训练格式
        training_data = []
        for item in generated:
            training_data.append({
                "conversations": [
                    {"role": "user", "content": item["input"]},
                    {"role": "assistant", "content": json.dumps(item["output"], ensure_ascii=False)}
                ]
            })

        return training_data

    except Exception as e:
        print(f"解析失败: {e}")
        return []

# 生成各意图数据
all_data = []

intents_config = {

            {"input": "查一下杭州的天气", "output": {"intent": "query_weather", "slots": {"location": "杭州", "date": "今天"}}},

}

# 生成并保存
with open('dataset/lora_intent.jsonl', 'w', encoding='utf-8') as f:
```

### 步骤 3: 数据验证

```python
import json

def validate_intent_data(file_path):
    """验证意图识别数据"""

    errors = []
    stats = {"intents": {}, "total": 0}

    with open(file_path, 'r', encoding='utf-8') as f:
        for line_num, line in enumerate(f, 1):
            try:
                data = json.loads(line)

                # 检查 conversations
                if 'conversations' not in data:
                    errors.append(f"行{line_num}: 缺少 conversations")
                    continue

                convs = data['conversations']
                if len(convs) < 2:
                    errors.append(f"行{line_num}: 对话不足2条")
                    continue

                # 提取意图
                output = convs[-1]['content']
                try:
                    output_json = json.loads(output)
                    intent = output_json.get('intent', 'unknown')

                    # 统计
                    stats['intents'][intent] = stats['intents'].get(intent, 0) + 1
                    stats['total'] += 1

                except json.JSONDecodeError:
                    errors.append(f"行{line_num}: 输出不是合法JSON")

            except Exception as e:
                errors.append(f"行{line_num}: {e}")

    # 输出结果
    print("数据验证结果:")
    print(f"  总数据量: {stats['total']}")
    print(f"  意图分布:")
    for intent, count in stats['intents'].items():
        print(f"    {intent}: {count} ({count/stats['total']*100:.1f}%)")

    if errors:
        print(f"\n❌ 发现 {len(errors)} 个错误:")
        for err in errors[:10]:
            print(f"  {err}")
    else:
        print("\n✅ 数据验证通过！")

validate_intent_data('dataset/lora_intent.jsonl')
```

#### 2. 创建训练脚本

创建 `trainer/train_intent_lora.py`：

```python
import os
import sys
import argparse
import torch
import torch.nn as nn
from torch.utils.data import DataLoader
from transformers import AutoTokenizer
from model.model_minimind import MiniMindConfig, MiniMindForCausalLM
from model.model_lora import apply_lora, get_lora_state_dict
from dataset.lm_dataset import SFTDataset

# 参数
parser = argparse.ArgumentParser()
parser.add_argument("--base_model", type=str, default="../out/full_sft_512.pth")
parser.add_argument("--data_path", type=str, default="../dataset/lora_intent.jsonl")
parser.add_argument("--output_dir", type=str, default="../out/lora")
parser.add_argument("--lora_r", type=int, default=16)
parser.add_argument("--lora_alpha", type=int, default=32)
parser.add_argument("--batch_size", type=int, default=16)
parser.add_argument("--epochs", type=int, default=3)
parser.add_argument("--learning_rate", type=float, default=1e-4)
parser.add_argument("--max_seq_len", type=int, default=512)
parser.add_argument("--device", type=str, default="cuda")
args = parser.parse_args()

# 加载基础模型
print(f"加载基础模型: {args.base_model}")
tokenizer = AutoTokenizer.from_pretrained("./model/")
config = MiniMindConfig(hidden_size=512, num_hidden_layers=8)
model = MiniMindForCausalLM(config).to(args.device)
model.load_state_dict(torch.load(args.base_model, map_location=args.device))
model.eval()

# 应用 LoRA
print(f"应用 LoRA: r={args.lora_r}, alpha={args.lora_alpha}")
apply_lora(model, r=args.lora_r, lora_alpha=args.lora_alpha)

# 冻结非 LoRA 参数
for name, param in model.named_parameters():
    if "lora" not in name:
        param.requires_grad = False

# 准备数据
print(f"加载数据: {args.data_path}")
train_dataset = SFTDataset(args.data_path, tokenizer, max_length=args.max_seq_len)
train_loader = DataLoader(train_dataset, batch_size=args.batch_size, shuffle=True)

# 优化器
optimizer = torch.optim.AdamW(
    [p for p in model.parameters() if p.requires_grad],
    lr=args.learning_rate
)

# 训练循环
print(f"\n开始训练...")
for epoch in range(args.epochs):
    model.train()
    total_loss = 0

    for step, (input_ids, labels, loss_mask) in enumerate(train_loader):
        input_ids = input_ids.to(args.device)
        labels = labels.to(args.device)
        loss_mask = loss_mask.to(args.device)

        # 前向传播
        outputs = model(input_ids)
        logits = outputs.logits

        # 计算损失
        loss_fct = nn.CrossEntropyLoss(reduction='none')
        loss = loss_fct(logits.view(-1, logits.size(-1)), labels.view(-1))
        loss = loss.view(labels.size())
        loss = (loss * loss_mask).sum() / loss_mask.sum()

        # 反向传播
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        total_loss += loss.item()

        if step % 10 == 0:
            print(f"Epoch {epoch+1}/{args.epochs}, Step {step}, Loss: {loss.item():.4f}")

    avg_loss = total_loss / len(train_loader)
    print(f"Epoch {epoch+1} 完成, 平均 Loss: {avg_loss:.4f}")

# 保存 LoRA 权重
os.makedirs(args.output_dir, exist_ok=True)
output_path = f"{args.output_dir}/lora_intent_512.pth"
torch.save(get_lora_state_dict(model), output_path)
print(f"\n✅ LoRA 权重已保存: {output_path}")
```

### 3. 性能指标

计算准确率：

```python
def evaluate_accuracy(model, tokenizer, test_data_file):
    """计算意图识别准确率"""

    correct = 0
    total = 0
    intent_correct = 0
    slot_correct = 0

    with open(test_data_file, 'r', encoding='utf-8') as f:
        for line in f:
            data = json.loads(line)
            convs = data['conversations']

            # 真实标签
            expected_output = json.loads(convs[-1]['content'])
            expected_intent = expected_output['intent']
            expected_slots = expected_output['slots']

            # 模型预测
            messages = convs[:-1]  # 只输入，不给答案
            prompt = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
            inputs = tokenizer(prompt, return_tensors="pt").to("cuda")

            with torch.no_grad():
                outputs = model.generate(**inputs, max_new_tokens=256, temperature=0.1)
            response = tokenizer.decode(outputs[0][inputs["input_ids"].shape[1]:], skip_special_tokens=True)

            # 解析预测
            try:
                predicted = json.loads(response)
                predicted_intent = predicted.get('intent')
                predicted_slots = predicted.get('slots', {})

                # 意图准确率
                if predicted_intent == expected_intent:
                    intent_correct += 1
                    correct += 1

                # 槽位准确率
                if predicted_slots == expected_slots:
                    slot_correct += 1

                total += 1

            except json.JSONDecodeError:
                total += 1  # JSON解析失败

    # 输出结果
    print(f"\n评估结果:")
    print(f"  总样本数: {total}")
    print(f"  意图准确率: {intent_correct/total*100:.1f}%")
    print(f"  槽位完全匹配率: {slot_correct/total*100:.1f}%")
    print(f"  JSON格式正确率: {correct/total*100:.1f}%")

# 使用
evaluate_accuracy(model, tokenizer, 'dataset/test_intent.jsonl')
```

## 常见问题

### Q1: 意图准确率不理想怎么办？
A:
1. 增加训练数据（特别是错误样本）
2. 调整 LoRA 秩（r=32）
3. 数据增强（同义替换）
4. 使用规则引擎辅助

### Q2: JSON格式不稳定怎么办？
A:
1. 降低生成温度（temperature=0.1）
2. 使用约束解码（outlines库）
3. 增加格式样本
4. 后处理重试机制

### Q3: 如何处理未见过的意图？
A:
1. 增加 general_chat 类别兜底
2. 设置 confidence 阈值
3. 低confidence时请求澄清
4. 对接大模型API处理
