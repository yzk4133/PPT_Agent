#### 3. 数据收集脚本示例

```python
import json
import requests
from bs4 import BeautifulSoup

def scrape_wikipedia(title):
    """爬取维基百科文章"""
    url = f"https://zh.wikipedia.org/wiki/{title}"
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')

    # 提取正文内容
    content = soup.find('div', {'class': 'mw-parser-output'}).get_text()
    content = ' '.join(content.split())  # 清理空白

    return {"text": content}

# 使用示例
titles = ["人工智能", "机器学习", "深度学习"]
with open('pretrain_custom.jsonl', 'w', encoding='utf-8') as f:
    for title in titles:
        data = scrape_wikipedia(title)
        f.write(json.dumps(data, ensure_ascii=False) + '\n')
```

### 数据清洗

```python
import re
import json

def clean_text(text):
    """清洗预训练文本"""
    # 移除多余空白
    text = re.sub(r'\s+', ' ', text)

    # 移除特殊字符
    text = re.sub(r'[\x00-\x08\x0b-\x0c\x0e-\x1f\x7f-\x9f]', '', text)

    # 移除过短文本（<50字符）
    if len(text) < 50:
        return None

    # 截断过长文本（>4096字符）
    if len(text) > 4096:
        text = text[:4096]

    return text

# 清洗脚本
with open('pretrain_raw.jsonl', 'r', encoding='utf-8') as fin, \
     open('pretrain_clean.jsonl', 'w', encoding='utf-8') as fout:
    for line in fin:
        data = json.loads(line)
        text = clean_text(data['text'])
        if text:
            fout.write(json.dumps({"text": text}, ensure_ascii=False) + '\n')
```

```python
import json
from openai import OpenAI

client = OpenAI(api_key="your-api-key")

def generate_sft_pairs(topic, count=10):
    """使用 GPT/Qwen 生成对话数据"""
    prompt = f"""
    请生成 {count} 组关于"{topic}"的问答对话。
    格式要求：
    - 问题要多样化
    - 回答要准确、简洁
    - 返回JSON数组格式
    """

    response = client.chat.completions.create(
        model="qwen-plus",
        messages=[{"role": "user", "content": prompt}]
    )

    # 解析并保存
    conversations = json.loads(response.choices[0].message.content)
    with open('sft_generated.jsonl', 'a', encoding='utf-8') as f:
        for conv in conversations:
            f.write(json.dumps(conv, ensure_ascii=False) + '\n')

# 生成示例
generate_sft_pairs("人工智能", 100)
generate_sft_pairs("编程", 100)
generate_sft_pairs("健康", 100)
```

### LoRA数据准备脚本

```python
import json

def create_intent_data():
    """创建意图识别训练数据"""

    # 定义意图模板
    templates = [
        ("query_weather", ["帮我查{date}{location}的天气", "{location}{date}下雨吗", "看看{date}{location}天气如何"]),
        ("book_flight", ["订一张{date}{location}的机票", "帮我买{origin}到{destination}的票"]),
        ("set_alarm", ["设置{time}的闹钟", "在{time}提醒我"]),
    ]

    # 生成变体
    data = []
    for intent, patterns in templates:
        for pattern in patterns:
            # 填充槽位
            text = pattern.format(
                date="明天",
                location="北京",
                origin="上海",
                destination="深圳",
                time="早上7点"
            )

            # 生成对应的JSON
            response = {
                "intent": intent,
                "slots": {"date": "明天", "location": "北京"}
            }

            data.append({
                "conversations": [
                    {"role": "user", "content": text},
                    {"role": "assistant", "content": json.dumps(response, ensure_ascii=False)}
                ]
            })

    # 保存
    with open('lora_intent.jsonl', 'w', encoding='utf-8') as f:
        for item in data:
            f.write(json.dumps(item, ensure_ascii=False) + '\n')

create_intent_data()
```

### DPO数据生成策略

```python
import json
from openai import OpenAI

client = OpenAI(api_key="your-api-key")

def generate_dpo_pair(user_input):
    """生成一个 DPO 偏好对"""

    # 生成好的回答
    good_prompt = f"""
    用户问题：{user_input}

    请生成一个专业、友好、有帮助的回答。
    """
    good_response = client.chat.completions.create(
        model="gpt-4",
        messages=[{"role": "user", "content": good_prompt}]
    ).choices[0].message.content

    # 生成差的回答
    bad_prompt = f"""
    用户问题：{user_input}

    请生成一个不礼貌、没有帮助的回答。
    """
    bad_response = client.chat.completions.create(
        model="gpt-4",
        messages=[{"role": "user", "content": bad_prompt}]
    ).choices[0].message.content

    # 构造 DPO 数据
    return {
        "chosen": [
            {"role": "user", "content": user_input},
            {"role": "assistant", "content": good_response}
        ],
        "rejected": [
            {"role": "user", "content": user_input},
            {"role": "assistant", "content": bad_response}
        ]
    }

# 批量生成
with open('dpo_custom.jsonl', 'w', encoding='utf-8') as f:
    questions = [
        "如何减肥？",
        "怎么提高工作效率？",
        "推荐几本好书",
    ]
    for q in questions:
        dpo_pair = generate_dpo_pair(q)
        f.write(json.dumps(dpo_pair, ensure_ascii=False) + '\n')
```

## 数据清洗与质量提升

### 通用清洗流程

```python
import json
import re

def clean_dataset(input_file, output_file):
    """通用数据清洗"""

    with open(input_file, 'r', encoding='utf-8') as fin,
         open(output_file, 'w', encoding='utf-8') as fout:

        for line_num, line in enumerate(fin, 1):
            try:
                data = json.loads(line)

                # 检查必需字段
                if 'conversations' not in data:
                    continue

                # 清理每条对话
                cleaned_conv = []
                for conv in data['conversations']:
                    if 'role' not in conv or 'content' not in conv:
                        continue

                    # 清理内容
                    content = conv['content'].strip()
                    content = re.sub(r'\s+', ' ', content)  # 多余空白

                    # 跳过空内容
                    if not content:
                        continue

                    cleaned_conv.append({
                        'role': conv['role'],
                        'content': content
                    })

                # 至少包含一对对话
                if len(cleaned_conv) < 2:
                    continue

                # 检查角色交替
                roles = [c['role'] for c in cleaned_conv]
                if roles[0] != 'user':
                    continue

                # 写入清洗后的数据
                fout.write(json.dumps({
                    'conversations': cleaned_conv
                }, ensure_ascii=False) + '\n')

            except Exception as e:
                print(f"行 {line_num} 出错: {e}")
                continue

# 使用示例
clean_dataset('sft_raw.jsonl', 'sft_clean.jsonl')
```

### 去重

```python
import json
from hashlib import md5

def deduplicate(input_file, output_file):
    """去除重复数据"""

    seen = set()
    with open(input_file, 'r', encoding='utf-8') as fin,
         open(output_file, 'w', encoding='utf-8') as fout:

        for line in fin:
            # 计算哈希
            content_hash = md5(line.encode('utf-8')).hexdigest()

            if content_hash in seen:
                continue  # 跳过重复

            seen.add(content_hash)
            fout.write(line)

deduplicate('sft_with_dups.jsonl', 'sft_dedup.jsonl')
```

### 质量过滤

```python
def quality_filter(input_file, output_file,
                min_length=10,
                max_length=4096,
                max_ratio=0.5):  # 中文字符占比

    with open(input_file, 'r', encoding='utf-8') as fin,
         open(output_file, 'w', encoding='utf-8') as fout:

        for line in fin:
            data = json.loads(line)

            # 计算总长度
            total_text = ''.join([c['content'] for c in data['conversations']])
            text_len = len(total_text)

            # 长度过滤
            if text_len < min_length or text_len > max_length:
                continue

            # 中文字符占比（过滤过多样英文）
            chinese_chars = len([c for c in total_text if '\u4e00' <= c <= '\u9fff'])
            if chinese_chars / text_len < max_ratio:
                continue

            fout.write(line)

quality_filter('sft_clean.jsonl', 'sft_filtered.jsonl')
```

## 数据格式验证

### 验证脚本

```python
import json
import sys

def validate_sft_data(file_path):
    """验证 SFT 数据格式"""

    errors = []

    with open(file_path, 'r', encoding='utf-8') as f:
        for line_num, line in enumerate(f, 1):
            try:
                # JSON 解析
                data = json.loads(line)

                # 检查 conversations 字段
                if 'conversations' not in data:
                    errors.append(f"行{line_num}: 缺少 conversations 字段")
                    continue

                convs = data['conversations']

                # 检查对话数量
                if len(convs) < 2:
                    errors.append(f"行{line_num}: 对话数量少于2条")

                # 检查每条对话
                for i, conv in enumerate(convs):
                    # 检查必需字段
                    if 'role' not in conv:
                        errors.append(f"行{line_num} 对话{i}: 缺少 role 字段")
                    if 'content' not in conv:
                        errors.append(f"行{line_num} 对话{i}: 缺少 content 字段")

                    # 检查 role 值
                    if conv.get('role') not in ['user', 'assistant', 'system']:
                        errors.append(f"行{line_num} 对话{i}: role 值无效")

                    # 检查内容为空
                    if not conv.get('content'):
                        errors.append(f"行{line_num} 对话{i}: content 为空")

            except json.JSONDecodeError as e:
                errors.append(f"行{line_num}: JSON 解析失败 - {e}")

    # 输出结果
    if errors:
        print(f"❌ 发现 {len(errors)} 个错误:")
        for error in errors[:10]:  # 只显示前10个
            print(f"  {error}")
        if len(errors) > 10:
            print(f"  ... 还有 {len(errors)-10} 个错误")
        return False
    else:
        print(f"✅ 数据格式验证通过！")
        return True

# 使用
validate_sft_data('sft_mini_512.jsonl')
```

### 数据统计

```python
def dataset_statistics(file_path):
    """统计数据集信息"""

    total_lines = 0
    total_chars = 0
    role_distribution = {'user': 0, 'assistant': 0}
    length_distribution = []

    with open(file_path, 'r', encoding='utf-8') as f:
        for line in f:
            data = json.loads(line)
            total_lines += 1

            for conv in data['conversations']:
                content = conv['content']
                total_chars += len(content)
                role_distribution[conv['role']] += 1
                length_distribution.append(len(content))

    # 输出统计
    avg_length = total_chars / (role_distribution['user'] + role_distribution['assistant'])

    print(f"数据集统计:")
    print(f"  总对话数: {total_lines}")
    print(f"  总字符数: {total_chars}")
    print(f"  平均长度: {avg_length:.1f} 字符")
    print(f"  用户消息: {role_distribution['user']}")
    print(f"  助手消息: {role_distribution['assistant']}")
    print(f"  最短: {min(length_distribution)} 字符")
    print(f"  最长: {max(length_distribution)} 字符")

dataset_statistics('sft_mini_512.jsonl')
```

- [ ] 包含 chosen 和 rejected
- [ ] 偏好差异明显
- [ ] 数据量足够（>1000对）

## 常见问题

### Q1: 需要多少数据？
A:
- 预训练: 1-10GB
- SFT: 0.5-5GB
- LoRA: 100-2000条即可

### Q2: 数据质量 vs 数量？
A: 质量优先！1000条高质量数据 > 10000条低质量数据。

### Q3: 如何平衡不同类别的数据？
A: 确保每个类别至少100-200条，避免类别不平衡。

### Q4: 可以混合中英文数据吗？
A: 可以，但建议主次分明。如中文为主，英文为辅（10-20%）。
